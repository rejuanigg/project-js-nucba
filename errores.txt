const newsContainer = document.querySelector(".news-p");
const seeMoreBtn = document.querySelector(".btn-load")

const createNewsTemplate = (news) => {
    const {id, title, newsImg, footer, categories} = news;
    return `
       <div class="news-box">
            <img src=${newsImg}>
            <div class="news-box-info">
                <h4>${title}</h4>
                <p>${footer}</p>
                    <div class="news-options">
                    <button class="btn-see-more btn-news"
                    data-id="${id}" 
                    data-title="${title}" 
                    data-newsImg="${newsImg}">Ver Más</button>
                    <button class="btn-add-notice btn-news"
                    data-id="${id}" 
                    data-title="${title}" 
                    data-newsImg="${newsImg}">+</button>
                    </div>
            </div>
       </div>
    `
}

const renderNews = (newsList) => {
    newsContainer.innerHTML += newsList.map(createNewsTemplate).join("");
};

const isLastIndexOf = () => {
    return appState.currentNewsIndex === appState.newsLimit-1;
}


const seeMoreNews = () => {
    appState.currentNewsIndex+=1;
    let {news, currentNewsIndex} = appState;
    renderNews(news[currentNewsIndex]);
    if (isLastIndexOf()){
        seeMoreBtn.classList.add("hidden");
    }
}

const init = () => {
    renderNews(appState.news[appState.currentNewsIndex]);
    seeMoreBtn.addEventListener("click", seeMoreNews)
};

init();
